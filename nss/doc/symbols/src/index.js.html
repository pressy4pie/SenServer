150   num_nodes = nodeCollection.count();

152     for(var i=1; i <= parseInt(value); i++){
153     thisNode = nodeCollection.find( { _id: parseInt(i) }).toArray();

156       node_to_check_str = JSON.stringify(node_to_check);
157       // Get rid of the brackets. 
158       node_to_check_str = node_to_check_str.replace(/[\[\]']+/g,'');
159       
160       // Convert it back. Thanks mongodb.
161       var node_json = JSON.parse(node_to_check_str);
162       
163       // If node hasnt been seen in the node_dead_milis

167         nodeCollection.update( {'_id' : parseInt(node_json._id)}, {$set: {'alive': false}} );

385 /** @summary Create a String/MS packet. 
386  *  @desc Create a String/MS packet. 
387  *  @param {string} destination - the node that will recieve this message. 
388  *  @param {string} sensor - the sensor on that node.
389  *  @param {string} command - which internal type. See mymessage.js.
390  *  @param {string} acknowledge - ack. 0 or 1.
391  *  @param {string} type - see mymessage.js.
392  *  @param {string} payload -the payload of the message.
393  *  @returns {string} msg - a MySensors packet.
394  */
395 function ms_encode(destination, sensor, command, acknowledge, type, payload) {
396 	var msg = destination.toString(10) + ";" + sensor.toString(10) + ";" + command.toString(10) + ";" + acknowledge.toString(10) + ";" + type.toString(10) + ";";
397 	if (command == 4) {
398 		for (var i = 0; i < payload.length; i++) {
399 			if (payload[i] < 16)
400 				msg += "0";
401 			msg += payload[i].toString(16);
402 		}
403 	} else {
404 		msg += payload;
405 	}
406 	msg += '\n';
407 	return msg.toString();
408 }

450 /** A proper mysensors packet has been recieved.  
451  * @param {object} _data - a mysensors packet.
452 */
453 function packet_recieved(_data){
454   // Split the raw data by semicolon.
455   var packet = _data.toString().trim().split(";");
456   
457   // Rip the packet apart by semicolon as noted above.
458   var nodeid = parseInt(packet[0]);
459   var childsensorid = parseInt(packet[1]);
460   var messagetype = parseInt(packet[2]);
461   var ack = parseInt(packet[3]);
462   var subtype = parseInt(packet[4]);
463   var payload = packet[5]; //This is a string most of the time..
464   
465   // Which type of message?
466   switch (parseInt(messagetype)) {
467     case C_PRESENTATION:
468       // this is a bug, motor controllers broadcast as 0 at first for some reason. Just ignore this..
469       if (childsensorid == "0" || nodeid == "0") break;
470       
471       // presentation is on broadcast address.
472       if (childsensorid == BROADCAST_ADDRESS){
473         // The version of the sensor. 
474         save_node_version(nodeid,payload.trim());
475       }
476       else if (childsensorid != BROADCAST_ADDRESS){
477         save_sensor(nodeid, childsensorid, payload,subtype);
478       }
479       else {
480         // uncaught presentation. Im not perfect ok.
481         logUtils.log('something weird?: ' + _data);
482       } 
483       break;
484 
485     case C_SET:
486       // this is a bug, motor controllers broadcast as 0 at first for some reason. 
487       if (childsensorid == "0" || nodeid == "0") break;
488       save_sensor_value(nodeid, childsensorid, subtype, payload);
489       break;
490 
491     case C_REQ:
492       logUtils.mslog('C_REQ message');
493       // I think time messages might go in here.
494       break;
495       
496     case C_INTERNAL:
497     // Catch that bug in motor controllers.
498     //if (childsensorid == "0" || nodeid == "0") break;
499       switch(parseInt(subtype)){
500         
501         // The sensors battery has been updated.                  
502         case I_BATTERY_LEVEL:
503           save_node_battery_level(nodeid,payload);
504           break;
505           
506         // Sensor is requesting time. 
507         case I_TIME:
508           ms_sendtime(nodeid);
509           break;
510           
511         case I_ID_REQUEST:
512           logUtils.mslog('ID request.');
513           if ( inclusion_mode == true ){ sendNextAvailableSensorId(); }
514           else logUtils.mslog('NOT IN INCLUSION MODE'); 
515           break;
516           
517         // Trigger inclusion_mode for a while
518         case I_INCLUSION_MODE:
519           toggle_inclusion_mode(parseInt(payload.trim()));
520           break;
521           
522         // Send the config to the node.   
523         case I_CONFIG:
524           ms_sendconfig(nodeid);
525           break;
526           
527         // We dont use this but we could. 
528         case I_LOG_MESSAGE:
529           if (nodeid == "0") break;
530           logUtils.mslog("LOG MESSAGE FROM: " + nodeid + " MESSAGE: " + payload.trim() );
531           break;
532           
533         // Node name and version. 
534         case I_SKETCH_NAME:
535           if (nodeid == "0") break;
536           save_node_name(nodeid, payload);
537           break;
538         case I_SKETCH_VERSION:
539           if (nodeid == "0") break;
540           save_node_lib_version(nodeid,payload);
541           break;
542           
543         // The gateway is ready. 
544         case I_GATEWAY_READY:
545           logUtils.mslog("Serial Gateway Ready.");
546           global.gateway_ready = true;
547           break;
548           
549         // i know the heartbeat is usefull but not using it yet. 
550         case I_HEARTBEAT:
551           logUtils.mslog('heartbeat: ' + _data);
552           break;
553         case I_HEARTBEAT_RESPONSE:
554           logUtils.mslog('heartbeat Response: ' + _data);
555           break;
556         
557         // I dont think we use the rest of these in tis situation.   
558         case I_PRESENTATION:
559           logUtils.mslog('presentation message: '+ _data);
560           break;
561           
562         // We don't use any of these. 
563         case I_VERSION:
564         break;
565         case I_ID_RESPONSE:
566         break;  
567         case I_FIND_PARENT:
568         break;
569         case I_FIND_PARENT_RESPONSE:
570         break;  
571         case I_CHILDREN:
572         break;
573         case I_SIGNING_PRESENTATION:
574         break;
575         case I_NONCE_REQUEST:
576         break;
577         case I_NONCE_RESPONSE:
578         break;
579         case I_REBOOT:
580         break;
581         case I_CHANNEL:
582         break;
583         case I_LOCKED:
584         break;  
585         case I_DISCOVER:
586         break;
587         case I_DISCOVER_RESPONSE:
588         break; 
589       }
590       break;
591       
592     case C_STREAM:
593       logUtils.mslog('C_STREAM message');
594       switch(parseInt(subtype)){
595         case ST_FIRMWARE_CONFIG_REQUEST:
596           break; 
597         case ST_FIRMWARE_CONFIG_RESPONSE:
598           break;
599         case ST_FIRMWARE_REQUEST:
600           break;
601         case ST_FIRMWARE_RESPONSE:
602           break;
603         case ST_SOUND:
604           break;
605         case ST_IMAGE:
606           break;
607         } 
608       break;
609   }

644 mqtt_client.on('message', function (topic, message) {
645   switch (topic){
646     
647     // If its a plain subscription to the base level, publish all nodes,alarms,timers etc.
648     case '/zc/' + serial_number + "/":
649       logUtils.mqttlog('new connection');
650       publish_all();
651       mqtt_client.publish('/zc/' + serial_number + "/get_current_inclusion_mode/",'get');
652       break;
653    
654    // This is just a test that i use for stuff.    
655     case '/zc/' + serial_number + "/test/":
656       break;
657       
658     /*
659     * INCLUSION MODE STUFFS.
660     */
661     
662     // User is able to start and end inclusion_mode with the app. 
663     case '/zc/' + serial_number + "/set_inclusion_mode/":
664       toggle_inclusion_mode(1);
665       mqtt_client.publish('/zc/' + serial_number + "/current_inclusion_mode/", 'on');
666       
667       setTimeout(function(){ 
668         mqtt_client.publish('/zc/' + serial_number + "/stop_inclusion_mode/", '');
669       }, 3000);
670       break;
671     
672     // end inclusion mode. 
673     case '/zc/' + serial_number + "/stop_inclusion_mode/":
674       toggle_inclusion_mode(0);
675       mqtt_client.publish('/zc/' + serial_number + "/current_inclusion_mode/", 'off');
676       break; 
677       
678     case '/zc/' + serial_number + "/get_current_inclusion_mode/":
679       if( inclusion_mode == true){
680         mqtt_client.publish('/zc/' + serial_number + "/current_inclusion_mode/", 'on');
681       } else {
682         mqtt_client.publish('/zc/' + serial_number + "/current_inclusion_mode/", 'off');
683       }
684       
685       break;     
686  
687     /*
688     * SENSOR STUFFS.
689     */
690       
691     /** Update a sensor variable. */
692     case '/zc/' + serial_number + '/update_sensor_variable/':
693       msg_json = JSON.parse(message.toString());
694       msg = ms_encode(msg_json['node_id'], msg_json['sensor_id'], msg_json['msg_cmd'], 0, msg_json['msg_type'], msg_json['payload'] );
695       ms_write_msg(msg);
696       break; 
697 
698     // Publish all the nodes. 
699     case '/zc/' + serial_number + '/get_nodes/':
700       publish_nodes();
701       break;      
702       
703     /*
704     * TIMER STUFFS.
705     */ 
706     
707     // Create a new timer. 
708     case '/zc/' + serial_number + '/create_timer/':
709       msg_json = JSON.parse(message.toString());
710       save_timer( msg_json );
711       break;
712 
713     // Publish timers.
714     case '/zc/' + serial_number + '/get_timers/':
715       publish_timers();
716       break;
717       
718     /*
719     * ALARM STUFFS.
720     */
721     
722     // Create an alarm. 
723     case '/zc/' + serial_number + '/create_alarm/':
724       msg_json = JSON.parse(message.toString());
725       save_alarm( msg_json );
726       break;
727 
728     // Publish alarms. 
729     case '/zc/' + serial_number + '/get_alarms/':
730       publish_alarms();
731       break;
732       
733     /*
734     * DEBUG STUFFS. 
735     */
736     case '/zc/' + serial_number + '/debug/raw_ms_msg/':
737       break;
738       
739     // Print the message if it doesnt match anything else. 
740     default:
741       break;
742   }
743 });
